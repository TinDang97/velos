       pass.set_bind_group(0, &self.agent_bind_group, &[]);
       // LOD selection based on camera_focus_distance:
       // < 500m: use full extrusion buffers
       // 500-1500m: use flat footprint buffers
       // > 1500m: skip draw entirely
       let (vbuf, ibuf, count) = if camera_focus_distance < 500.0 {
